#version 430 core
layout (location = 1) out vec2 out_t;

const vec2 pos[4] = vec2[4](vec2(-1.0, 1.0), vec2(1.0, 1.0), vec2(-1.0, -1.0), vec2(1.0, -1.0));
const vec2 tex[4] = vec2[4](vec2(0, 1), vec2(1, 1), vec2(0, 0), vec2(1, 0));
const int ind[6] = int[6](0, 1, 2, 2, 1, 3);

//uniform mat4x4 WVP;

void main() {
    int vi = ind[gl_VertexID % 6];
    gl_Position = vec4(pos[vi], 0, 1);
    out_t = tex[vi];
}
